Example 1

1. read data from a csv file and show the head

import pandas as pd
df = pd.read_csv("./../data/meteorite-landings.csv")
df.head()

2. Save the data in variables (dataframes)

df2 = df[df["reclat"].notnull()]
df3 = df2[df2["reclong"].notnull()]
long = df3["reclong"]
lat = df3["reclat"]

3. Show the relevant data in a picture/map/card

%matplotlib inline
import matplotlib.pyplot as plt
import cartopy.crs as ccrs
import cartopy.feature as cfeature

fig = plt.figure(figsize=(15, 20))
ax = plt.axes(projection=ccrs.Robinson())

ax.set_global()
ax.coastlines(resolution="50m")

ax.plot(long, lat, linestyle=" ", marker="X", transform=ccrs.PlateCarree(), color=(0, 0, 0), alpha=0.2)

ax.add_feature(cfeature.OCEAN, zorder=0)
ax.add_feature(cfeature.LAND, zorder=0)
ax.add_feature(cfeature.BORDERS, zorder=0, edgecolor='gray')
ax.gridlines()
plt.show()

Example 2

1. Create a form with input/select fields
and save the data in an output file.

import csv
import ipywidgets as widgets
from IPython.display import display

firstname = widgets.Text(description="Firstname:", value="")
display(firstname)

lastname = widgets.Text(description="Lastname:", value="")
display(lastname)

subject_options = [
"---",
"Science",
"Technology",
"Engineering",
"Mathematics"
]

subject = widgets.Dropdown(description="Subject:", options=subject_options)
display(subject)

button = widgets.Button(description="Save")
display(button)

def button_click(event):
firstname_value = firstname.value
lastname_value = lastname.value
subject_value = subject.value

if firstname_value != "" and lastname_value != "" and subject_value != "---":
with open(".students.csv", "a", newline="", encoding="utf-8") as f:
writer = csv.writer(f, delimiter=";")
writer.writerow([firstname_value, lastname_value, subject_value])
print("Data saved in file")
firstname.value = ""
lastname.value = ""
subject.value = "---"
else:
print("Please fill all fields with values")

button.on_click(button_click)
