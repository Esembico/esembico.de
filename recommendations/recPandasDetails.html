Example 2

Read the file and show how often the name was given

name = "Anna" #search value 1
gender = "F" #search value 2
state = "CA" #search value 3

import pandas as pd #import pandas
df = pd.read_csv("../data/names.csv") #read the data
df.head() #show the head for the orientation

df2 = df[df["Name"] == name] #first filter
df3 = df2[df2["Gender"] == gender] #second filter
df4 = df3[df3["State"] == state] #third filter
df5 = df4.sort_values("Year") #sort the data

%matplotlib inline
import matplotlib.pyplot as plt #import matplotlib

plt.plot(df5["Year"], df5["Count"]) #show the data
